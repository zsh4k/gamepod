class Gamepod {
	/**
	 * @param {number} responseTime Min 1 max 500
	 */
	constructor(responseTime = 50) {
		//config
		this._intervalId = undefined;
		this.gamepads = [];
		this.gamepad = 0;

		//Events
		this.onJoystickMove = undefined;
		this.onPress = undefined;
		this.onTrigger = undefined;
		this.onAction = undefined;

		//gamelopp
		this._gameloop = () => {};

		window.addEventListener("gamepadconnected", () => {
			this.gamepads = navigator.getGamepads();
			this.gamepad = this.gamepads.length;
			this._intervalId = setInterval(this._gameloop, responseTime);
		});

		window.addEventListener("gamepaddisconnected", (ev) => {
			clearInterval(this._intervalId);
		});
	}

	/**
	 * @description Run when move the Joysticks
	 * @param {(event: {})=> void} callback Run when move the Joysticks
	 */
	set onJoystickMove(callback) {}
}
